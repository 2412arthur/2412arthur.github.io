def fenetre_decode() :

    fenetre_2 = Tk()
    fenetre_2.title("Hidden Message - Décoder un message")
    fenetre_2['bg'] = '#55b8d1'

    cadre_2 = Frame(fenetre_2, width=768, height=576,  borderwidth=2, relief=GROOVE)
    cadre_2.pack(fill=BOTH, side=TOP, padx=5, pady=5)

    filepath = askopenfilename(title="Ouvrir une image",
                                filetypes=[('png files','.png'),('all files','.*')])
    photo = PhotoImage(master = cadre_2, file=filepath)
    canvas = Canvas(cadre_2, width=photo.width(), height=photo.height(), bg="yellow")
    canvas.create_image(0, 0, anchor=NW, image=photo)
    canvas.pack()

    bouton_retour_1 = Button(fenetre_2, text="Retour au menu", command=fenetre_2.destroy)
    bouton_retour_1.pack(fill=X, side=BOTTOM, padx=5, pady=5)

    fenetre_2.mainloop()
